import React, { useState } from 'react';
import { Search, Sparkles, Sidebar as Shuffle, Users, BookOpen } from 'lucide-react';
 
const Hero = ({ onSearch, onRecommend }) => {
    const [query, setQuery] = useState('');
    const [results, setResults] = useState([]);
 
    const handleSearchChange = async (e) => {
        const val = e.target.value;
        setQuery(val);
        if (val.length > 2) {
            try {
                const resp = await fetch(`/api/search_books?query=${val}`);
                const data = await resp.json();
                setResults(data);
            } catch (err) {
                console.error(err);
            }
        } else E{
            setResults([]);
        }
    };
 
    return (
        <section className="py-20 flex flex-col items-center text-center px-4">
            <div className="bg-blue-500/10 text-blue-400 px-4 py-1 rounded-full text-sm font-semibold mb-6 flex items-center gap-2 border border-blue-500/20">
                <Sparkles size={16} />
                AI-Powered Recommendations
            </div>
 
            <h1 className="text-5xl md:text-7xl font-extrabold mb-6 bg-gradient-to-r from-blue-400 to-purple-500 bg-clip-text text-transparent">
                BookSage-AI
            </h1>
            <p className="text-xl text-slate-400 max-w-2xl mb-10 leading-relaxed">
                Discover your next favorite book with our intelligent hybrid recommendation engine
                that learns from millions of readers
            </p>
 
            <div className="w-full max-w-2xl relative mb-8">
                <div className="relative group">
                    <input
                        type="text"
                        className="w-full bg-slate-800/50 border border-slate-700 rounded-2xl py-5 px-6 pl-14 focus:outline-none focus:ring-2 focus:ring-blue-500/50 transition-all text-lg shadow-2xl"
                        placeholder="Search for a book you love..."
                        value={query}
                        onChange={handleSearchChange}
                    />
                    <Search className="absolute left-5 top-1/2 -translate-y-1/2 text-slate-500 group-focus-within:text-blue-500 transition-colors" />
                    <button className="absolute right-4 top-1/2 -translate-y-1/2 bg-blue-600 hover:bg-blue-500 p-3 rounded-xl transition-all shadow-lg active:scale-95">
                        <Search size={20} className="text-white" />
                    </button>
                </div>
 
                {results.length > 0 && (
                    <div className="absolute top-full left-0 w-full mt-2 bg-slate-800 border border-slate-700 rounded-xl shadow-2xl z-50 overflow-hidden">
                        {results.map((book, i) => (
                            <button
                                key={i}
                                className="w-full flex items-center gap-4 p-4 hover:bg-slate-700/50 transition-all border-b border-slate-700 last:border-0 text-left"
                                onClick={() => {
                                    setQuery(book.title);
                                    setResults([]);
                                }}
                            >
                                <img src={book.image_url} alt={book.title} className="w-10 h-14 object-cover rounded" />
                                <div>
                                    <div className="font-semibold text-slate-200">{book.title}</div>
                                    <div className="text-sm text-slate-400">{book.author}</div>
                                </div>
                            </button>
                        ))}
                    </div>
                )}
            </div>
 
            <div className="flex flex-wrap justify-center gap-4">
                <button
                    onClick={() => onRecommend(query, 'hybrid')}
                    className="flex items-center gap-2 px-8 py-3 bg-gradient-to-r from-blue-600 to-blue-500 hover:from-blue-500 hover:to-blue-400 rounded-xl font-bold transition-all shadow-lg active:scale-95"
                >
                    <Shuffle size={20} />
                    Hybrid
                </button>
                <button
                    onClick={() => onRecommend(query, 'collaborative')}
                    className="flex items-center gap-2 px-8 py-3 bg-slate-800 hover:bg-slate-700 border border-slate-700 rounded-xl font-bold transition-all shadow-lg active:scale-95"
                >
                    <Users size={20} />
                    Collaborative
                </button>
                <button
                    onClick={() => onRecommend(query, 'content')}
                    className="flex items-center gap-2 px-8 py-3 bg-slate-800 hover:bg-slate-700 border border-slate-700 rounded-xl font-bold transition-all shadow-lg active:scale-95"
                >
                    <BookOpen size={20} />
                    Content-Based
                </button>
            </div>
        </section>
    );
};
 
export default Hero;
